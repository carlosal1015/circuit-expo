\providecommand{\interval}[1]{
	\directlua{
		tex.print(
		"\string\\textbf{Tiempo estimado}: " .. #1 .. " segundos."..
		"\string\\hfill" .. "$t\string\\in" .. "\string\\left[" ..
				trunc(start/60,2) .. "," .. difference(#1) ..
				"\string\\right]" .. "$" .. " min.")
	}
}
